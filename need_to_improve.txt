import os
import json
from openai import OpenAI
from pharma_triage_env.env import PharmaTriageEnv

API_BASE_URL = os.getenv("API_BASE_URL") or "https://router.huggingface.co/v1"
API_KEY = os.getenv("HF_TOKEN") or os.getenv("API_KEY")
MODEL_NAME = os.getenv("MODEL_NAME")

client = OpenAI(base_url=API_BASE_URL, api_key=API_KEY)

SYSTEM_PROMPT = """
Return ONLY JSON:
{
  "severity": "low | medium | high",
  "serious": true | false,
  "expected": true | false,
  "escalation": "routine_review | urgent_review | regulatory_report"
}
"""

def call_model(obs):
    try:
        res = client.chat.completions.create(
            model=MODEL_NAME,
            temperature=0,
            messages=[
                {"role": "system", "content": SYSTEM_PROMPT},
                {"role": "user", "content": json.dumps(obs)}
            ]
        )
        return json.loads(res.choices[0].message.content)
    except:
        return fallback(obs)

def fallback(obs):
    serious = obs.get("hospitalized") or obs.get("life_threatening")
    expected = bool(set(obs.get("symptoms", [])) & set(obs.get("known_label_side_effects", [])))

    return {
        "severity": "high" if serious else "medium",
        "serious": serious,
        "expected": expected,
        "escalation": "regulatory_report" if serious and not expected else "urgent_review" if serious else "routine_review"
    }

def run(task):
    env = PharmaTriageEnv(task=task)
    total = 0

    for _ in range(30):
        obs = env.reset()
        action = call_model(obs)
        _, reward, _, _ = env.step(action)
        total += max(0, min(1, reward / 10))

    return total / 30

def main():
    scores = {t: run(t) for t in ["easy", "medium", "hard"]}
    print(scores)
    print("overall:", sum(scores.values()) / 3)

if __name__ == "__main__":
    main()
